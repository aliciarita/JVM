53 byteTres = u1Read(fp);
54 byteDois = u1Read(fp);
55 byteUm = u1Read(fp);

63 destino = ((byteTres)<<24) | ((byteDois)<<16) | ((byteUm)<<8) | ((byteZero));
64 /*Caso contrário, retorna um valor de debug*/
