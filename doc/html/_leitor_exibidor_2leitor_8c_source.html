759double decodificaDoubleInfo (cp_info * cp) {
760 uint64_t valor = ((uint64_t)cp->UnionCP.CONSTANT_Double.high_bytes<<32) | (uint64_t)cp->UnionCP.CONSTANT_Double.low_bytes;
761 int sinal = ((valor>>63) == 0) ? 1 : -1;
762 int expon = ((valor>>52) & 0x7ffL);
763 long mant = (expon == 0) ? ((valor & 0xfffffffffffffL) << 1) : ((valor & 0xfffffffffffffL) | 0x10000000000000L);
764
765 double retorno = sinal*mant*(pow(2,expon-1075));
766 return retorno;
767}

786 return retorno;
787}
